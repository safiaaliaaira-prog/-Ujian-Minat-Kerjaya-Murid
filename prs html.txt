    let currentIdx = 0;
    let scores = { sains: 0, sastera: 0, teknik: 0 };
    let quizQuestions = [];

    function initQuiz() {
        quizQuestions = [...allQuestions].sort(() => Math.random() - 0.5);
        currentIdx = 0;
        scores = { sains: 0, sastera: 0, teknik: 0 };
        document.getElementById('result-area').style.display = 'none';
        document.getElementById('quiz-card').style.display = 'block';
        updateQuizUI();
    }

    function updateQuizUI() {
        const q = quizQuestions[currentIdx];
        const progress = ((currentIdx / quizQuestions.length) * 100);
        document.getElementById('q-count').innerText = `Soalan ${currentIdx + 1}/${quizQuestions.length}`;
        document.getElementById('q-percent').innerText = Math.round(progress) + '%';
        document.getElementById('progress-fill').style.width = progress + '%';
        document.getElementById('question-text').innerText = q.q;

        const options = document.getElementById('option-container');
        options.innerHTML = `
            <button class="btn btn-yes" onclick="handleAnswer(true)" style="background:#00b894;color:white;">YA, SANGAT MINAT</button>
            <button class="btn btn-no" onclick="handleAnswer(false)" style="background:#ff7675;color:white;">TIDAK / KURANG MINAT</button>
        `;
    }

    function handleAnswer(val) {
        if(val) scores[quizQuestions[currentIdx].field]++;
        currentIdx++;
        if(currentIdx < quizQuestions.length) updateQuizUI();
        else showResult();
    }

    function showResult() {
        document.getElementById('quiz-card').style.display = 'none';
        document.getElementById('result-area').style.display = 'block';

        let winner = Object.keys(scores).reduce((a, b) => scores[a] > scores[b] ? a : b);
        const resHeader = document.getElementById('res-header');
        const resTitle = document.getElementById('res-title');
        const resDesc = document.getElementById('res-desc');

        if(winner === 'sains') {
            resHeader.style.background = 'linear-gradient(135deg, #00d2ff, #3a7bd5)';
            resTitle.innerText = "ALIRAN SAINS";
            resDesc.innerText = "Minda awak sangat analitikal! Awak sesuai dalam bidang perubatan atau kejuruteraan.";
        } else if(winner === 'sastera') {
            resHeader.style.background = 'linear-gradient(135deg, #ff9a9e, #fecfef)';
            resTitle.innerText = "ALIRAN SASTERA";
            resDesc.innerText = "Awak sangat kreatif dan berjiwa kemanusiaan tinggi.";
        } else {
            resHeader.style.background = 'linear-gradient(135deg, #a18cd1, #fbc2eb)';
            resTitle.innerText = "ALIRAN TEKNIKAL";
            resDesc.innerText = "Awak pakar dalam kerja amali dan kemahiran tangan.";
        }

        document.getElementById('s-score').innerText = scores.sains;
        document.getElementById('st-score').innerText = scores.sastera;
        document.getElementById('t-score').innerText = scores.teknik;

        saveLocal(resTitle.innerText);
    }

    function saveLocal(aliran) {
        const history = JSON.parse(localStorage.getItem('saf_aliran_v2') || "[]");
        history.unshift({ aliran, date: new Date().toLocaleDateString('ms-MY') });
        localStorage.setItem('saf_aliran_v2', JSON.stringify(history.slice(0, 5)));
        renderHistory();
    }

    function renderHistory() {
        const history = JSON.parse(localStorage.getItem('saf_aliran_v2') || "[]");
        const container = document.getElementById('history-container');
        container.innerHTML = history.length === 0 ? "<p style='font-size: 0.75rem; color: #999;'>Belum ada rekod.</p>" : 
            history.map(h => `<div class="history-row"><span><strong>${h.aliran}</strong></span><span>${h.date}</span></div>`).join('');
    }

    window.onload = renderHistory;
